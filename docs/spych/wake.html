 37        Usage:

 39        - Checks whether this listener should stop processing and exit early
 40        - Resets `kill` and `locked` state if stopping is required
 41
 42        Returns:
 43

 55    def __call__(self):
 56        """
 57        Usage:
 58
 59        - Executes one full listen-and-detect cycle when this listener is invoked as a thread target
 60        - Records audio, transcribes it, and triggers a wake event if any wake word is detected
 61
 62        Notes:
 63
 64        - Skips execution silently if this listener is already locked (i.e. mid-cycle)
 65        - Checks `should_stop` at each major step to allow early exit without blocking
 66        - Uses `beam_size=2` for fast transcription appropriate for short wake word clips
 67        - The `initial_prompt` biases the model toward all registered wake words to reduce
 68          false negatives
 69        - If multiple wake words are present in a single segment, the first match wins
 70        """
 71        if self.locked:
 72            self.notify(
 73                "Listener is locked, skipping...", notification_type="verbose"
 74            )
 75            return
 76        if self.should_stop():
 77            return
 78        self.locked = True
 79        buffer = record(
 80            device_index=self.spych_wake_object.device_index,
 81            duration=self.spych_wake_object.wake_listener_time,
 82        )
 83        if self.should_stop():
 84            return
 85        audio_buffer = get_clean_audio_buffer(buffer)
 86        if self.should_stop():
 87            return
 88        wake_words = list(self.spych_wake_object.wake_word_map.keys())
 89        segments, _ = self.spych_wake_object.wake_model.transcribe(
 90            audio_buffer,
 91            beam_size=2,
 92            initial_prompt=f"You are listening for wake words: {str(wake_words)}. Do not transcribe anything else. You should be quite sure you hear these words before you return them.",
 93        )
 94        for segment in segments:
 95            if self.should_stop():
 96                return
 97            text = segment.text.lower()
 98            for wake_word in wake_words:
 99                if wake_word in text:
100                    self.spych_wake_object.wake(wake_word)
101                    self.locked = False
102                    self.kill = False
103                    return
104        self.locked = False
105        self.kill = False

109    def __init__(
110        self,
111        wake_word_map,
112        terminate_words=None,
113        wake_listener_count=3,
114        wake_listener_time=2,
115        wake_listener_max_processing_time=0.5,
116        device_index=-1,
117        whisper_model="tiny",
118        whisper_device="cpu",
119        whisper_compute_type="int8",
120    ):
121        """
122        Usage:
123
124        - Initializes a wake word detection system using overlapping listener threads
125          and faster-whisper for offline transcription
126        - Supports multiple wake words, each mapped to a different callback function
127
128        Requires:
129
130        - `wake_word_map`:
131            - Type: dict[str, callable]
132            - What: A dictionary mapping wake words to their corresponding no-argument
133              callback functions
134            - Note: Keys are stored and matched in lowercase
135            - Example:
136                {
137                    "jarvis": on_jarvis_wake,
138                    "computer": on_computer_wake,
139                }
140
141        Optional:
142
143        - `terminate_words`:
144            - Type: list[str]
145            - What: A list of words that, if detected in the wake listener's transcription,
146              will immediately terminate the entire SpychWake system
147            - Note: Use with caution, as any false positive on a terminate word will stop
148              the wake system until it is manually restarted
149            - default: None (disabled)
150
151        - `wake_listener_count`:
152            - Type: int
153            - What: The number of concurrent listener threads to run
154            - Default: 3
155            - Note: More listeners reduce the chance of missing a wake word between
156              recording windows; at least 3 is recommended for continuous coverage
157
158        - `wake_listener_time`:
159            - Type: int | float
160            - What: The duration in seconds each listener records per cycle
161            - Default: 2
162
163        - `wake_listener_max_processing_time`:
164            - Type: int | float
165            - What: The estimated maximum time in seconds for transcription to complete
166            - Default: 0.5
167            - Note: Used alongside `wake_listener_time` and `wake_listener_count` to
168              calculate the stagger delay between thread launches
169
170        - `device_index`:
171            - Type: int
172            - What: The microphone device index to record from
173            - Default: -1
174            - Note: Use `-1` to select the system default input device
175
176        - `whisper_model`:
177            - Type: str
178            - What: The faster-whisper model name to use for wake word transcription
179            - Default: "tiny"
180            - Note: Smaller models (tiny, base) are recommended here for low latency
181
182        - `whisper_device`:
183            - Type: str
184            - What: The device to run the whisper model on
185            - Default: "cpu"
186            - Note: Use "cuda" for GPU acceleration if available
187
188        - `whisper_compute_type`:
189            - Type: str
190            - What: The compute type to use for the whisper model
191            - Default: "int8"
192            - Note: "int8" offers a good balance of speed and accuracy on both CPU and GPU
193        """
194        self.wake_word_map = {k.lower(): v for k, v in wake_word_map.items()}
195        # Handle Terminating Words
196        self.terminate_words = (
197            [w.lower() for w in terminate_words] if terminate_words else []
198        )
199        for word in self.terminate_words:
200            if word in self.wake_word_map:
201                raise ValueError(
202                    f"Terminate word '{word}' cannot also be a wake word."
203                )
204            self.wake_word_map[word] = self.stop
205        self.wake_listener_count = wake_listener_count
206        self.wake_listener_time = wake_listener_time
207        self.wake_listener_max_processing_time = (
208            wake_listener_max_processing_time
209        )
210        self.device_index = device_index
211        self.locked = False
212        self.kill = False
213        self.wake_model = WhisperModel(
214            whisper_model,
215            device=whisper_device,
216            compute_type=whisper_compute_type,
217        )
218        self.wake_listeners = [
219            SpychWakeListener(self) for _ in range(self.wake_listener_count)
220        ]
221
222    def start(self):
223        """
224        Usage:
225
226        - Starts the wake word detection loop using overlapping listener threads
227        - Blocks until a KeyboardInterrupt is received or `stop()` is called
228
229        Notes:
230
231        - Callbacks are defined in `wake_word_map` at init time rather than passed to `start`
232        - Listener threads are staggered by `(wake_listener_time + wake_listener_max_processing_time)
233          / wake_listener_count` seconds to ensure continuous audio coverage
234        - New threads are only launched when the system is not locked (i.e. not currently
235          processing a wake event)
236        """
237        self.notify(
238            f"Listening for wake words: {list(self.wake_word_map.keys())}...",
239            notification_type="verbose",
240        )
241        try:
242            while True:
243                for listener in self.wake_listeners:
244                    if self.kill:
245                        self.kill = False
246                        return
247                    if not self.locked:
248                        threading.Thread(target=listener).start()
249                    time.sleep(
250                        (
251                            self.wake_listener_time
252                            + self.wake_listener_max_processing_time
253                        )
254                        / self.wake_listener_count
255                    )
256        except KeyboardInterrupt:
257            self.notify("Stopping.", notification_type="verbose")
258
259    def stop_listeners(self):
260        """
261        Usage:
262
263        - Signals all listener threads to stop at their next available checkpoint
264        - Note: Does not block; listeners will exit cleanly after their current operation
265        """
266        for listener in self.wake_listeners:
267            listener.stop()
268
269    def stop(self):
270        """
271        Usage:
272
273        - Stops all listener threads and exits the `start` loop
274        - Note: Combines `stop_listeners` with setting the kill flag on the main loop
275        """
276        self.stop_listeners()
277        self.kill = True

279    def wake(self, wake_word):
280        """
281        Usage:
282
283        - Called internally when a wake word is detected
284        - Stops all listeners, locks the system, executes the mapped callback for the
285          detected wake word, then unlocks
286
287        Requires:
288
289        - `wake_word`:
290            - Type: str
291            - What: The detected wake word, used to look up the correct callback in
292              `wake_word_map`
293
294        Notes:
295
296        - If the system is already locked when `wake` is called, the call is a no-op
297          to prevent concurrent wake executions
298        - Any exception raised by the callback is caught and re-raised as a spych exception
299        - The system is always unlocked in the `finally` block, even if the callback raises
300        """
301        self.stop_listeners()
302        if self.locked:
303            return
304        self.locked = True
305        try:
306            self.wake_word_map[wake_word]()
307        except Exception as e:
308            self.notify(
309                f"Error in on_wake_fn for '{wake_word}': {e}",
310                notification_type="exception",
311            )
312        finally:
313            self.locked = False

223    def start(self):
224        """
225        Usage:
226
227        - Starts the wake word detection loop using overlapping listener threads
228        - Blocks until a KeyboardInterrupt is received or `stop()` is called
229
230        Notes:
231
232        - Callbacks are defined in `wake_word_map` at init time rather than passed to `start`
233        - Listener threads are staggered by `(wake_listener_time + wake_listener_max_processing_time)
234          / wake_listener_count` seconds to ensure continuous audio coverage
235        - New threads are only launched when the system is not locked (i.e. not currently
236          processing a wake event)
237        """
238        self.notify(
239            f"Listening for wake words: {list(self.wake_word_map.keys())}...",
240            notification_type="verbose",
241        )
242        try:
243            while True:
244                for listener in self.wake_listeners:
245                    if self.kill:
246                        self.kill = False
247                        return
248                    if not self.locked:
249                        threading.Thread(target=listener).start()
250                    time.sleep(
251                        (
252                            self.wake_listener_time
253                            + self.wake_listener_max_processing_time
254                        )
255                        / self.wake_listener_count
256                    )
257        except KeyboardInterrupt:
258            self.notify("Stopping.", notification_type="verbose")

281        """
282        Usage:
283
284        - Called internally when a wake word is detected
285        - Stops all listeners, locks the system, executes the mapped callback for the
286          detected wake word, then unlocks
287
288        Requires:
289

311                notification_type="exception",
312            )
313        finally:
314            self.locked = False
